const copyPasswordToClipboard = useCallback(() => {
  const passwordInput = passwordRef.current;
  if (passwordInput) {
    passwordInput.select();
    document.execCommand('copy'); // Copy the selected text to clipboard
  }
}, []);

// Assuming passwordRef is created using useRef and attached to your input element
const passwordRef = useRef();

// In your JSX, attach the ref to the input element
<input ref={passwordRef} type="text" value={password} readOnly />

// The button to trigger the copy function
<button onClick={copyPasswordToClipboard}>Copy Password</button>
